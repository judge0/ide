const apiUrl = "https://judge0-ce.p.rapidapi.com/about";  // Your API endpoint

// Fetch data from the API
function fetchData() {
    fetch(apiUrl, {
        method: 'GET', // Use POST if you need to send data
        headers: {
            'Content-Type': 'application/json',
            // 'Authorization': 'Bearer YOUR_API_KEY' // Uncomment if you need authentication
        }
    })
    .then(response => response.json())  // Parse the JSON response
    .then(data => {
        console.log("Data received:", data); // Log the data you receive from the API
        // Further processing of the data can be done here
    })
    .catch(error => {
        console.error("Error fetching data:", error); // Handle any errors that occur during the fetch
    });
}

// Call the function to fetch the data
fetchData();
